import 'package:http/http.dart' as http;

void main() {
  var urls = [
    "http://httpbin.org/delay/2", // 2 second delay
    "http://httpbin.org/delay/1", // 1 second delay
  ];

  print("Starting sequential chain...");

  http.get(Uri.parse(urls[0])).then((response1) {
    print("First response: [Data for ${urls[0]}]");
    
    return http.get(Uri.parse(urls[1]));

  }).then((response2) {
    print("Second response: [Data for ${urls[1]}]");

  }).then((_) {
    print("Chain complete.");
  });

  print("End of main");
}

import 'package:http/http.dart' as http;

void main() {
  var urls = [
    "http://httpbin.org/delay/3", // 3 sec
    "http://httpbin.org/delay/1", // 1 sec
    "http://httpbin.org/delay/2", // 2 sec
  ];

  List<Future<void>> responseFutures = [];

  print("Starting all requests in parallel...");

  for (var url in urls) {
    print("Queueing fetch for $url");
    var future = http.get(Uri.parse(url)).then((response) {
      print("COMPLETED: $url [Data for $url]");
    });
    responseFutures.add(future);
  }
  Future.wait(responseFutures).then((_) {
    print("--- All requests are finished. ---");
  });

  print("End of main");
}
